# 文档验证脚本

## 检查文档格式

```python
import zipfile
import xml.etree.ElementTree as ET
import re

def check_docx_format(docx_path):
    """检查docx文档格式是否符合软著要求"""
    
    with zipfile.ZipFile(docx_path, 'r') as docx_zip:
        # 检查样式定义
        with docx_zip.open('word/styles.xml') as styles_xml:
            tree = ET.parse(styles_xml)
            root = tree.getroot()
            
            ns = {'w': 'http://schemas.openxmlformats.org/wordprocessingml/2006/main'}
            
            # 查找Normal样式
            normal_style = root.find('.//w:style[@w:styleId="Normal"]', ns)
            if normal_style is not None:
                rPr = normal_style.find('w:rPr', ns)
                if rPr is not None:
                    sz = rPr.find('w:sz', ns)
                    if sz is not None:
                        size_val = sz.get('{http://schemas.openxmlformats.org/wordprocessingml/2006/main}val')
                        print(f"正文字号: {size_val} (半磅) = {int(size_val)/2}磅")
            
            # 检查Heading1样式
            heading1_style = root.find('.//w:style[@w:styleId="Heading1"]', ns)
            if heading1_style is not None:
                rPr = heading1_style.find('w:rPr', ns)
                if rPr is not None:
                    sz = rPr.find('w:sz', ns)
                    if sz is not None:
                        size_val = sz.get('{http://schemas.openxmlformats.org/wordprocessingml/2006/main}val')
                        print(f"Heading1字号: {size_val} (半磅) = {int(size_val)/2}磅")
        
        # 检查文档内容
        with docx_zip.open('word/document.xml') as doc_xml:
            tree = ET.parse(doc_xml)
            root = tree.getroot()
            
            ns = {'w': 'http://schemas.openxmlformats.org/wordprocessingml/2006/main'}
            
            # 统计段落
            paragraphs = root.findall('.//w:p', ns)
            print(f"段落总数: {len(paragraphs)}")
            
            # 统计标题
            headings = []
            for para in paragraphs:
                pPr = para.find('w:pPr', ns)
                if pPr is not None:
                    pStyle = pPr.find('w:pStyle', ns)
                    if pStyle is not None:
                        style_val = pStyle.get('{http://schemas.openxmlformats.org/wordprocessingml/2006/main}val')
                        if style_val and 'heading' in style_val.lower():
                            text = ''.join(node.text or '' for node in para.findall('.//w:t', ns))
                            headings.append((style_val, text))
            
            print(f"标题总数: {len(headings)}")
            
            # 统计图片
            images = root.findall('.//wp:inline', ns)
            print(f"内嵌图片数量: {len(images)}")
            
            # 统计表格
            tables = root.findall('.//w:tbl', ns)
            print(f"表格数量: {len(tables)}")
            
            # 检查关键词
            all_text = ''.join(node.text or '' for node in root.findall('.//w:t', ns))
            keywords = ['架构图', '数据流向图', '技术栈图', '功能模块图']
            print("\n配图标题检查:")
            for kw in keywords:
                if kw in all_text:
                    print(f"  ✓ 找到 '{kw}'")
                else:
                    print(f"  ✗ 未找到 '{kw}'")

# 使用示例
# check_docx_format('说明书.docx')
```

## 检查信息采集表

```python
def check_info_form(docx_path):
    """检查信息采集表必填字段"""
    
    with zipfile.ZipFile(docx_path, 'r') as docx_zip:
        with docx_zip.open('word/document.xml') as doc_xml:
            tree = ET.parse(doc_xml)
            root = tree.getroot()
            
            ns = {'w': 'http://schemas.openxmlformats.org/wordprocessingml/2006/main'}
            
            # 获取所有文本
            all_text = ''.join(node.text or '' for node in root.findall('.//w:t', ns))
            
            # 检查必填字段
            required_fields = ['软件全称', '版本号', '开发完成日期', '著作权人', '证件号码']
            print("必填字段检查:")
            for field in required_fields:
                if field in all_text:
                    print(f"  ✓ 找到 '{field}'")
                else:
                    print(f"  ✗ 未找到 '{field}'")
            
            # 检查表格
            tables = root.findall('.//w:tbl', ns)
            print(f"\n表格数量: {len(tables)}")
            
            # 显示表格内容
            for t, table in enumerate(tables):
                rows = table.findall('.//w:tr', ns)
                print(f"\n表格 {t+1} ({len(rows)} 行):")
                for r, row in enumerate(rows[:5]):  # 只显示前5行
                    cells = row.findall('.//w:tc', ns)
                    cell_texts = []
                    for cell in cells:
                        cell_text = ''.join(node.text or '' for node in cell.findall('.//w:t', ns))
                        cell_texts.append(cell_text.strip())
                    print(f"  行{r+1}: {cell_texts}")

# 使用示例
# check_info_form('信息采集表.docx')
```

## 统计文档字数

```python
def count_docx_stats(docx_path):
    """统计docx文档字数和段落数"""
    
    with zipfile.ZipFile(docx_path, 'r') as docx_zip:
        with docx_zip.open('word/document.xml') as doc_xml:
            tree = ET.parse(doc_xml)
            root = tree.getroot()
            
            ns = {'w': 'http://schemas.openxmlformats.org/wordprocessingml/2006/main'}
            
            # 统计段落
            paragraphs = root.findall('.//w:p', ns)
            
            # 统计字数
            total_chars = 0
            for para in paragraphs:
                text = ''.join(node.text or '' for node in para.findall('.//w:t', ns))
                total_chars += len(text)
            
            print(f"段落总数: {len(paragraphs)}")
            print(f"总字数: {total_chars}")
            print(f"平均字数/段落: {total_chars / len(paragraphs) if paragraphs else 0:.1f}")

# 使用示例
# count_docx_stats('说明书.docx')
```
